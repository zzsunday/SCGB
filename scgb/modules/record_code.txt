def NoName(path, purity):
    data1 = load_json(path)
    #print(data1.keys())
    
    drop_bc = defaultdict(lambda: defaultdict())
    filter_bc = defaultdict(lambda: defaultdict(int))
    
    # 预先构建 `Barcode` 到 `Purity` 和 `ReadNum` 的映射字典，避免循环内查询 DataFrame
    purity_dict = purity.set_index('Barcode')[['Purity', 'ReadNum']].to_dict(orient='index')
    
    # 遍历 `data1`
    for contig, barcodes_dict in data1.items():  # 用 `list()` 以支持 `pop()` 操作
        #print(contig)
        for bc in barcodes_dict.keys():  # 用 `list()` 以支持 `pop()` 操作
            if bc in purity_dict:  # 确保 `bc` 在 `purity_dict` 中
                purity_value = purity_dict[bc]['Purity']
                #print(purity_value)
                read_num = float(purity_dict[bc]['ReadNum'])
                #exp_reads = float(read_num * purity_value * 0.005/100)
                exp_reads = max(5,float(read_num * purity_value * 0.005/100))
                #print(f"{bc} purtiy is {purity_value} and read num is {read_num}")
                if purity_value <= 99 and purity_value >= 80:

                    #print(f"{bc} exp_read is {exp_reads}")
                    if barcodes_dict[bc] < exp_reads:
    
                        drop_bc[contig][bc] = [barcodes_dict[bc], exp_reads, purity_value, read_num]
                    else:
                        #if bc=="TCACATTGCTATAGAGT":
                           # print(data1[contig][bc], purity)
                        filter_bc[contig][bc] = data1[contig][bc]
                           
                elif purity_value >= 100:
                    #if bc=="TCACATTGCTATAGAGT":
                    #       print(data1[contig][bc], purity_value)
                    #print(path, contig)
                    filter_bc[contig][bc] = data1[contig][bc]
                else:
                    drop_bc[contig][bc] = [barcodes_dict[bc], exp_reads, purity_value, read_num]
    return data1, filter_bc, drop_bc

def build_bin_to_plasmids(ref_df: pd.DataFrame):
    """Return: dict[bin_name] -> list of plasmid headers."""
    # normalize columns
    for col in ("Plasmid", "Bin"):
        if col not in ref_df.columns:
            raise ValueError(f"Missing column: {col}")
    # We will match bins by genome stem (filename without suffix) OR by filename
    ref_df = ref_df.copy()
    ref_df["Bin_norm"] = ref_df["Bin"].astype(str).str.replace(r"\s+", "", regex=True)
    d = (
        ref_df.groupby("Bin_norm")["Plasmid"]
        .apply(lambda x: [str(i) for i in x.dropna().tolist()])
        .to_dict()
    )
    return d

def make_plasmid_index(fasta_path: Path):
    """
    Index the plasmid FASTA by:
      - full record.id
      - token before first whitespace in header
    We provide a tolerant accessor to find records by header.
    """
    # Primary index by record.id (up to first whitespace)
    idx = SeqIO.index(str(fasta_path), "fasta")  # keys are record.id (tokenized)
    return idx

def find_plasmid_record(idx, key: str):
    """
    Try to fetch plasmid by several key variants:
      - exact key
      - key up to first whitespace
      - key up to first pipe '|'
    """
    # exact
    if key in idx:
        return idx[key]
    # up to whitespace
    k_ws = key.split()[0]
    if k_ws in idx:
        return idx[k_ws]
    # up to pipe
    k_pipe = key.split("|")[0]
    if k_pipe in idx:
        return idx[k_pipe]
    # sometimes headers look like '>gi|...|ref|XYZ|'; try last token
    parts = key.split("|")
    if parts:
        cand = parts[-1].split()[0]
        if cand in idx:
            return idx[cand]
    return None

def write_out(genome_path: Path, records_to_append, out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    out_fp = out_dir / genome_path.name
    #print(out_fp)
    # Write genome first (verbatim), then append plasmids
    with open(out_fp, "wb") as w:
        # copy genome file bytes directly to preserve headers/order
        with open(genome_path, "rb") as r:
            w.write(r.read())
        # ensure a newline between genome and appended plasmids
    if records_to_append:
        with open(out_fp, "a", encoding="utf-8", newline="\n") as w_txt:
            SeqIO.write(records_to_append, w_txt, "fasta")

    return out_fp

def main(ref: pd.DataFrame):
    bin2plas = build_bin_to_plasmids(ref) # return bin to list of plasmid header
    plasmid_idx = make_plasmid_index(PLASMID_DB) # get all plasmid headers

    genomes = sorted(GENOME_DIR.glob(f"*{GENOME_SUFFIX}")) # all bacteria chromosome fasta
    if not genomes:
        logger.warning(f"No genomes found under {GENOME_DIR} with suffix {GENOME_SUFFIX}")

    # To match: try both stem and filename
    for g in genomes:
        print(g)
        stem = g.stem
        filename = g.name
        keys_to_try = {stem, stem + GENOME_SUFFIX}

        matched_bin_key = None
        for k in keys_to_try:
            k_norm = k.replace(" ", "")
            if k_norm in bin2plas:
                matched_bin_key = k_norm
                break

        if matched_bin_key is None:
            # No plasmids: write genome only
            out_fp = write_out(g, [], OUT_DIR)
            logger.info(f"[NO PLASMID] Wrote genome only -> {out_fp.name}")
            continue

        # Collect plasmid SeqRecords
        plasmid_headers = bin2plas[matched_bin_key] # get plasmid headers
        records = []
        missing = []
        for hdr in plasmid_headers:
            rec = find_plasmid_record(plasmid_idx, hdr)
            if rec is None:
                missing.append(hdr)
            else:
                records.append(rec)

        if missing:
            logger.warning(f"{g.name}: {len(missing)} plasmid header(s) not found in DB: {missing[:5]}{'...' if len(missing)>5 else ''}")

        out_fp = write_out(g, records, OUT_DIR)
        logger.info(f"[WITH PLASMID] Wrote genome + {len(records)} plasmid(s) -> {out_fp.name}")

    # Close index
    plasmid_idx.close()
